def fifo_page_replacement(reference, frames):
    memory = []
    page_faults = 0
    index = 0

    print("FIFO Simulation")
    for page in reference:
        if page not in memory:
            page_faults += 1
            if len(memory) < frames:
                memory.append(page)
            else:
                memory[index] = page
                index = (index + 1) % frames
            status = "Fault"
        else:
            status = "Hit"
        print(f"Page {page} -> {memory} ({status})")

    return page_faults


def lru_page_replacement(reference, frames):
    memory = []
    recent = {}
    page_faults = 0

    print("\nLRU Simulation")
    for i, page in enumerate(reference):
        if page not in memory:
            page_faults += 1
            if len(memory) < frames:
                memory.append(page)
            else:
                lru_page = min(recent, key=recent.get)
                memory[memory.index(lru_page)] = page
                recent.pop(lru_page)
            status = "Fault"
        else:
            status = "Hit"

        recent[page] = i
        print(f"Page {page} -> {memory} ({status})")

    return page_faults


if __name__ == "__main__":
    reference = [7, 0, 1, 2, 0, 3, 0, 4, 2, 3, 0, 3, 2]
    frames = 3

    fifo_faults = fifo_page_replacement(reference, frames)
    lru_faults = lru_page_replacement(reference, frames)

    print("\nSummary")
    print(f"FIFO Page Faults: {fifo_faults}")
    print(f"LRU Page Faults : {lru_faults}")
